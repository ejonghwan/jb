# 설치

// 글로벌(g) 설치
npm install -g typescript

// 해당 폴더(로컬) 설치
npm install typescript


# 체크

// 글로벌
tsc -v

// 로컬
npx tsc -v


# config 생성

// 글로벌 (글로벌의 경우 해당 폴더로 가는 거 잊지 말자)
tsc --init

// 로컬
npx tsc --init
