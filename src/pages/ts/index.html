    interface User {
       name: string,
       age: number
    }
    
    
    // age를 넘버 혹은 문자로 입력받고 그에 따라 객체or문자를 리턴해줄 때 
    function join(name: string, age: number | string): User | string {
       if(typeof age === 'number') {
           return {
               name, age
           }
       } else {
           return "나이는 숫자로 입력"
       }

    사용할 때 다양한 타입지정 
    
    
    ex1) 사용이유
    function getSize(arr: number[]): number {
        return arr.length;
    }
    
    getSize([1,2,3]) //숫자로만 전달 시 에러가 나지않음
    getSize([1,2,'d']) //하지만 문자가 섞여있으면 에러 
    
    
    // 해결방법은 유니온타입 or 오버로드 사용 
    function getSize(arr: number[] | string[]): number {
        return arr.length;
    }
    
    
    ex2) // 사용방법
    //하지만 타입이 여러개면 코드가 길고 복잡해짐. !!그때 사용하는게 제네릭

    function getSize<T>(arr: T[]): number {
        return arr.length;
    }
    
    getSize<number>([1,2,3])
    getSize<number | string>([1,2,'d'])
